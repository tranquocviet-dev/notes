\subsubsection{Identification}
\label{sec:org9316b94}
\begin{itemize}
\item Based on the available information on the coursework specifications, there are 3 already defined Entities, namely Customer, Item, and Suppliers.
\item For better management of the addresses of Living entities groups, the Addresses will be stored in seperate Address Entities
\item As per the requirement of the coursework, the database has to furthermore support the online sales management, with
framework available to expand with delivery integration and customer service
\begin{itemize}
\item For sale management, a Sale and a Receipt entity will be integrated to monitor the items being brought
\item For delivery integration, a Delivery Service entity will be created, along with its seperate Address entry
\end{itemize}
\end{itemize}

\subsection{3rd Normalization Form Process}
\label{sec:orge3cf6f9}
\begin{itemize}
\item Customer Table
\begin{itemize}
\item 1st Normal Form: No Redundant Data (No Duplicate Values): The Customer Table has no duplicate values and no values as part of another value set
\item 2nd Normal Form: Clear Data Relationships: The Customer table has clear relationships with other tables
\item 3rd Normal Form: Remove Anomalies: The Customer Table has no partial dependencies, and no stray values from other tables.
\end{itemize}
\item Item Table
\begin{itemize}
\item 1st Normal Form: No Redundant Data (No Duplicate Values): The Item Table has no duplicate values and no values as part of another value set
\item 2nd Normal Form: Clear Data Relationships: The Item table has clear relationships with other tables
\item 3rd Normal Form: Remove Anomalies: The Item Table has no partial dependencies, and no stray values from other tables.
\end{itemize}
\item Supplier Table

\subsubsection{Source code for the table:}
\label{sec:org62efe78}
\begin{center}
\begin{verbatim}
CREATE TABLE Zip
(
    ZipCode VARCHAR(128) NOT NULL PRIMARY KEY,
    Country VARCHAR(128),
    Region VARCHAR(128),
    City VARCHAR(128)1
);
\end{verbatim}
\begin{verbatim}
CREATE TABLE CustomerAddress
(
    CustomerAddressID INT NOT NULL PRIMARY KEY,
    CustomerAddressStreet VARCHAR(128),
    CustomerAddressStreetNum INT(128),
    CustomerAddressZip VARCHAR(128),
    FOREIGN KEY (CustomerAddressZip) REFERENCES Zip(ZipCode)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE DeliveryAddress
(
    DeliveryAddressID INT NOT NULL PRIMARY KEY,
    DeliveryAddressStreet VARCHAR(128),
    DeliveryAddressStreetNum INT(128),
    DeliveryAddressZip VARCHAR(128),
    FOREIGN KEY (DeliveryAddressZip) REFERENCES Zip(ZipCode)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE SupplierAddress
(
    SupplierAddressID INT NOT NULL PRIMARY KEY,
    SupplierAddressStreet VARCHAR(128),
    SupplierAddressStreetNum INT(128),
    SupplierAddressZip VARCHAR(128),
    FOREIGN KEY (SupplierAddressZip) REFERENCES Zip(ZipCode)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE Customer
(
    CustomerID INT NOT NULL PRIMARY KEY,
    CustomerName VARCHAR(128),
    CustomerPhone INT(128),
    CustomerAddressIDCode INT,
    FOREIGN KEY (CustomerAddressIDCode) REFERENCES CustomerAddress(CustomerAddressID)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE Supplier
(
    SupplierID INT NOT NULL PRIMARY KEY,
    SupplierName VARCHAR(128),
    SupplierPhone INT(128),
    SupplierAddressIDCode INT,
    FOREIGN KEY (SupplierAddressIDCode) REFERENCES SupplierAddress(SupplierAddressID)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE Item
(
    ItemID INT NOT NULL PRIMARY KEY,
    ItemName VARCHAR(128),
    ItemSupplierID INT(128),
    ItemCategory VARCHAR(128),
    ItemLeft INT(128),
    ItemCost INT(128),
    CONSTRAINT SupplierID FOREIGN KEY (ItemSupplierID)
    REFERENCES Supplier(SupplierID)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE DeliveryService
(
    DeliveryServiceID INT NOT NULL PRIMARY KEY,
    DeliveryServiceName VARCHAR(128),
    DeliveryServicePhone INT(128),
    DeliveryAddressIDCode INT,
    FOREIGN KEY (DeliveryAddressIDCode) REFERENCES DeliveryAddress(DeliveryAddressID)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE Sale
(
    SaleID INT NOT NULL PRIMARY KEY,
    SaleCustomerID INT,
    SaleDeliveryServiceID INT,
    FOREIGN KEY (SaleCustomerID) REFERENCES Customer(CustomerID),
    FOREIGN KEY (SaleDeliveryServiceID) REFERENCES DeliveryService(DeliveryServiceID)
);

\end{verbatim}
\begin{verbatim}
CREATE TABLE Receipt
(
    ReceiptSaleID INT,
    ReceiptItemID INT,
    ReceiptItemCount INT(128),
    FOREIGN KEY (ReceiptSaleID) REFERENCES Sale(SaleID),
    FOREIGN KEY (ReceiptItemID) REFERENCES Item(ItemID)
);
\end{verbatim}
\end{center}
